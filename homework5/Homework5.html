import numpy as np
def formCircle(xc, yc, zc, r, s):
    theta= np.linspace(2 * 3.14, 0, s)             
    x, z= [], []
    for i in theta:
        x.append(r * sin(i) + xc)           
        z.append(r * cos(i) + zc)           
    y = np.ones(s) * yc
    return x, y, z
